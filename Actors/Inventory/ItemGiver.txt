Actor SmallHealthFusion : CustomInventory replaces SmallHealth //10001
{
//$Category MM8BDM-Health/Energy
-COUNTITEM
+INVENTORY.IGNORESKILL
Inventory.Amount 1
Inventory.MaxAmount 1
Inventory.PickUpMessage "Picked up a health capsule."
Inventory.PickUpSound "item/energyup"
Scale 2.0
States
{
Spawn:
HBAL AB 6
Loop
Pickup:
TNT1 A 0
TNT1 A 0 A_JumpIfinventory("PoisonMark",1, "Detox")
TNT1 A 0 A_JumpIf(ACS_ExecuteWithResult(GetTotalHPPerc,0)<100,"Healme") 
TNT1 A 0
Fail
Healme:
TNT1 A 0 HealThing(10) //ACS_ExecuteWithResult(GetValueHPPerc,15)
TNT1 A 0 A_GiveInventory("TempFireProof",2)
TNT1 A 0
stop
Detox:
TNT1 A 1 A_Giveinventory("DetoxLevel",25)
TNT1 A 1 A_Giveinventory("DetoxEfectSpawn")
stop
Nope:
TNT1 A 0 
Fail
}
}
